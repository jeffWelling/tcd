      def runTriggers
        PROFILES.each {|m|
          mod=m.to_s[MODULE_NAME_REGEX]
          TCD::Profiles.getInterfaces(mod).each {|interface|
            usage= percentOfCapacity mod, interface
            TCD::Triggers.update( mod, interface, usage )
          }
        }
      end
