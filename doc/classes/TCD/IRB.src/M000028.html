      def getAllProfileStats
        result={}
        PROFILES.each {|profile|
          stats_plus_timestamp=profile.getStats.merge({:timestamp=>Time.now})
          result.merge!({"#{profile}"[/[^:]+?$/].to_sym => stats_plus_timestamp}) if profile.useProfile?
        }
        result
      end
