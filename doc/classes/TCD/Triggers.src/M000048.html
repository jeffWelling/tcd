      def update profile_name, interface, percent
        profile_name=profile_name.to_sym
        interface=interface.to_sym
        
        @triggers[profile_name][interface][percent].each {|rules|
          eval(rules[1]) if eval(rules[0]).class==TrueClass
        } rescue return(false)
        true
      end
